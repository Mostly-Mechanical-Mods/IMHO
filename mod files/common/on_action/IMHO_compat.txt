on_game_start_after_lobby = {
	on_actions = {
		IMHO_compat
	}
}

IMHO_compat = {
	effect = {
		set_global_variable = {
			name = IMHO_is_loaded
			value = yes
		}
		set_global_variable = {
			name = IMHO_mod_version
			value = 1.1
		}
		set_global_variable = {
			name = IMHO_vanilla_version
			value = 1.9
		}
	}
}
